┌─ ПАРАМЕТРЫ ГЕНЕРАЦИИ ───────────────────────────────────────────────────────┐
│                                                                               │
│  DungeonConfig.asset (ScriptableObject)                                       │
│                                                                               │
│  Размеры карты:                                                               │
│  • mapWidth = 100                    (в тайлах)                              │
│  • mapHeight = 100                   (в тайлах)                              │
│                                                                               │
│  Размеры комнат:                                                              │
│  • minRoomSize = (8, 8)              (минимум 8x8 тайлов)                    │
│  • maxRoomSize = (20, 20)            (максимум 20x20 тайлов)                 │
│                                                                               │
│  Типы комнат:                                                                 │
│  • normalRoomsCount = 15             (комнаты с врагами)                     │
│  • specialRoomsCount = 2             (магазины, ловушки)                     │
│  • Start = 1 (всегда)                (стартовая комната)                     │
│  • Boss = 1 (всегда)                 (комната босса)                         │
│                                                                               │
│  Маршруты:                                                                    │
│  • minPathToBoss = 5                 (минимальное расстояние)                │
│  • maxPathToBoss = 10                (максимальное расстояние)               │
│  • branchProbability = 0.3           (вероятность веток 0.0-1.0)            │
│                                                                               │
│  Генерация:                                                                   │
│  • useRandomSeed = true              (новая карта каждый раз)                │
│  • seed = 0                          (для фиксированной карты)               │
│                                                                               │
│  Пример: Если изменить normalRoomsCount с 15 на 30                           │
│          - будет 30 обычных комнат вместо 15                                │
│          - карта будет плотнее                                                │
│          - больше врагов и лута                                               │
│                                                                               │
└───────────────────────────────────────────────────────────────────────────────┘

│  │ Может содержать: враги, лут          │                                    │
│  │ Цель: пройти и получить опыт         │                                    │
│  └────────────────────────────────────┘                                       │
│                                                                               │
│  ┌─ SPECIAL ──────────────────────────┐                                       │
│  │ Описание: Специальная комната        │                                    │

│  │ Может содержать: магазин, ловушки    │                                    │
│  │ Цель: решить головоломку или купить  │                                    │
│  └────────────────────────────────────┘                                       │
│                                                                               │
│  ┌─ BOSS ─────────────────────────────┐                                       │
│  │ Описание: Комната босса               │                                    │
│  │ Где находится: Всегда в конце        │                                    │
│  │ Может содержать: финальный враг      │                                    │
│  │ Цель: победить босса и закончить     │                                    │
